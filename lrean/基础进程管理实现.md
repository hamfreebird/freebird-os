const PAGE_SIZE: usize = 4096;

impl KernelStack {
    pub fn allocate(size: usize) -> Result<Self, ProcessError> {
        let layout = Layout::from_size_align(size, PAGE_SIZE)
            .map_err(|_| ProcessError::MemoryAllocationFailed)?;
        
        let ptr = unsafe { alloc_zeroed(layout) };
        if ptr.is_null() {
            return Err(ProcessError::MemoryAllocationFailed);
        }
        
        let bottom = VirtAddr::from_ptr(ptr).align_up(PAGE_SIZE);
        let top = bottom + size;
        Ok(KernelStack { bottom, top, size })
    }
    
    pub fn top(&self) -> VirtAddr { self.top }
    pub fn bottom(&self) -> VirtAddr { self.bottom }
    pub fn size(&self) -> usize { self.size }
}

pub fn process_exit(exit_code: i32) -> ! {
    let mut scheduler = SCHEDULER.lock();
    if let Some(current_pid) = scheduler.current_pid {
        if let Some(process) = scheduler.processes.get_mut(&current_pid) {
            process.exit_code = Some(exit_code);
            process.state = ProcessState::Terminated;
            println!("Process {} exited with code {}", current_pid.0, exit_code);
        }
    }
    scheduler.schedule();
    crate::hlt_loop();
}

## 系统架构总结

### **核心组件**
1. **进程控制块 (PCB)** - 存储进程所有状态信息
2. **调度器** - 管理进程状态转换和调度决策
3. **上下文切换** - 汇编实现的寄存器保存/恢复
4. **内存管理** - 内核栈分配和页表管理
5. **系统调用** - 用户空间与内核的接口

### **进程生命周期**
```
New → Ready ↔ Running ↔ Blocked → Terminated
      ↑        ↓          ↑
      └────────┴──────────┘
```

### **调度策略**
- **轮转调度**：公平时间片分配
- **优先级支持**：进程可设置优先级
- **空闲进程**：当没有就绪进程时运行

### **关键特性**
1. **抢占式调度**：通过定时器中断实现
2. **上下文隔离**：每个进程有独立寄存器状态
3. **错误处理**：完整的错误类型和恢复机制
4. **扩展性**：模块化设计便于添加新功能
