简单进程结构设计与实现

在操作系统中，进程是执行中的程序实例。一个简单的进程结构通常包含以下几个核心组件：

### 5. **调度器设计**

```rust
// src/process/scheduler.rs#L1-40
pub struct Scheduler {
    /// 就绪队列（按优先级分组）
    ready_queue: VecDeque<ProcessId>,
    /// 阻塞队列（等待I/O等）
    blocked_queue: VecDeque<ProcessId>,
    /// 所有进程的映射表
    processes: BTreeMap<ProcessId, Process>,
    /// 当前运行的进程
    current_pid: Option<ProcessId>,
    /// 空闲进程（当没有其他进程运行时执行）
    idle_process: ProcessId,
}

impl Scheduler {
    /// 创建新进程
    pub fn create_process(&mut self, entry_point: VirtAddr, name: &'static str) -> Result<ProcessId, ProcessError> {
        // 分配PID
        let pid = ProcessId::new();
        
        // 创建内核栈
        let kernel_stack = self.allocate_kernel_stack()?;
        
        // 初始化进程上下文
        let mut context = ProcessContext::new();
        context.rip = entry_point.as_u64();
        context.rsp = kernel_stack.top().as_u64();
        
        // 创建进程
        let process = Process {
            pid,
            state: ProcessState::Ready,
            context,
            kernel_stack,
            user_stack: None,
            page_table: Self::clone_current_page_table(),
            priority: 1,
            parent_pid: self.current_pid,
            exit_code: None,
            name,
            memory_regions: Vec::new(),
        };
        
        // 添加到进程表
        self.processes.insert(pid, process);
        self.ready_queue.push_back(pid);
        
        Ok(pid)
    }
    
    /// 调度下一个进程运行
    pub fn schedule(&mut self) -> Option<ProcessId> {
        if let Some(current_pid) = self.current_pid.take() {
            // 保存当前进程状态
            if let Some(process) = self.processes.get_mut(&current_pid) {
                if process.state == ProcessState::Running {
                    process.state = ProcessState::Ready;
                    self.ready_queue.push_back(current_pid);
                }
            }
        }
        
        // 选择下一个进程（简单轮转调度）
        if let Some(next_pid) = self.ready_queue.pop_front() {
            if let Some(process) = self.processes.get_mut(&next_pid) {
                process.state = ProcessState::Running;
                self.current_pid = Some(next_pid);
                return Some(next_pid);
            }
        }
        
        // 没有就绪进程，返回空闲进程
        Some(self.idle_process)
    }
}
```

### 6. **上下文切换汇编实现**

```asm
; src/process/switch.S#L1-40
global switch_context

; void switch_context(ProcessContext** current, ProcessContext* next)
switch_context:
    ; 保存当前上下文
    push rbp
    push rbx
    push r12
    push r13
    push r14
    push r15
    
    ; 保存当前栈指针到 current
    mov [rdi], rsp
    
    ; 加载下一个栈指针
    mov rsp, rsi
    
    ; 恢复下一个上下文
    pop r15
    pop r14
    pop r13
    pop r12
    pop rbx
    pop rbp
    
    ; 返回，此时会跳转到新进程的 rip
    ret
```

### 7. **内核栈管理**

```rust
// src/process/stack.rs#L1-30
pub struct KernelStack {
    /// 栈底（高地址）
    bottom: VirtAddr,
    /// 栈顶（低地址）
    top: VirtAddr,
    /// 栈大小
    size: usize,
}

impl KernelStack {
    /// 为进程分配内核栈
    pub fn allocate(size: usize) -> Result<Self, ProcessError> {
        // 从内核堆中分配内存
        let layout = Layout::from_size_align(size, PAGE_SIZE)
            .map_err(|_| ProcessError::MemoryAllocationFailed)?;
        
        let ptr = unsafe { alloc_zeroed(layout) };
        if ptr.is_null() {
            return Err(ProcessError::MemoryAllocationFailed);
        }
        
        let bottom = VirtAddr::from_ptr(ptr).align_up(PAGE_SIZE);
        let top = bottom + size;
        
        Ok(KernelStack { bottom, top, size })
    }
    
    /// 设置栈上的初始上下文
    pub fn setup_initial_context(&mut self, entry_point: VirtAddr) -> ProcessContext {
        // 在栈顶预留空间保存初始寄存器值
        let mut context = ProcessContext::new();
        context.rip = entry_point.as_u64();
        context.rsp = self.top.as_u64() - 8; // 预留返回地址空间
        context.rbp = context.rsp;
        context
    }
}
```

```rust
pub extern "C" fn syscall_handler(context: &mut ProcessContext) {
    let syscall_num = context.rax as u64;
    let arg1 = context.rbx;
    let arg2 = context.rcx;
    let arg3 = context.rdx;
    
    match syscall_num {
        n if n == SyscallNumber::Exit as u64 => {
            let exit_code = arg1 as i32;
            process_exit(exit_code);
        }
        n if n == SyscallNumber::GetPid as u64 => {
            let pid = current_process().pid.0;
            context.rax = pid;
        }
        // 处理其他系统调用...
        _ => {
            // 无效系统调用
            context.rax = !0;
        }
    }
```

### 9. **进程创建流程**

```rust
// src/process/builder.rs#L1-35
pub struct ProcessBuilder {
    entry_point: VirtAddr,
    name: &'static str,
    priority: u8,
    memory_size: usize,
    arguments: Vec<u64>,
    environment: Vec<&'static str>,
}

impl ProcessBuilder {
    /// 创建并启动新进程
    pub fn spawn(self) -> Result<ProcessId, ProcessError> {
        let scheduler = SCHEDULER.lock();
        
        // 1. 创建进程结构
        let pid = scheduler.create_process(self.entry_point, self.name)?;
        
        // 2. 分配内存空间
        let memory_space = MemorySpace::new(self.memory_size)?;
        
        // 3. 设置初始堆栈
        let stack = Stack::new()?;
        
        // 4. 复制参数到用户空间
        self.copy_arguments_to_user_space(pid)?;
        
        // 5. 设置入口点
        let process = scheduler.get_process_mut(pid).unwrap();
        process.context.rip = self.entry_point.as_u64();
        
        // 6. 添加到就绪队列
        scheduler.wake_process(pid);
        
        Ok(pid)
    }
}
```

## 设计要点总结

1. **分层设计**：上下文、内存、调度器分离，便于维护和扩展
2. **状态清晰**：明确的进程状态转换，避免竞态条件
3. **内存隔离**：每个进程有自己的地址空间（通过独立页表实现）
4. **抢占支持**：通过定时器中断实现时间片轮转调度
5. **系统调用**：提供用户空间与内核的受控接口
6. **资源管理**：跟踪进程使用的所有资源，确保完全释放

这种设计从简单开始，但预留了扩展空间，可以逐步添加：
- 用户/内核模式分离
- 进程间通信（IPC）
- 信号处理
- 文件描述符管理
- 多核调度

对于 Freebird OS，建议先从协作式多任务开始（进程主动 yield），然后逐步实现抢占式调度和内存保护。
